# 查看正在运行的容器

docker ps

# 启动容器

docker start 容器 id(CONTAINER ID)

# 重启容器

docker restart 容器 id(CONTAINER ID)

# 停止当前运行的容器

docker stop 容器 id(CONTAINER ID)

# 强制停止当前容器

docker kill 容器 id(CONTAINER ID)

# 查看主机的所有镜像

docker images

# 查看所有的容器

docker ps -a

# 拉取镜像

docker pull 镜像名:tag

# 构建镜像

docker build -t [ 镜像名 ]:[ tag ] [DockerFile 所在目录]

# 删除镜像

docker rmi 镜像名/镜像id

# 重启容器

docker restart 容器 id(CONTAINER ID)

# 多阶段构建例子(vue项目举例)

FROM node:latest as build-stage  # 拉去 node:latest镜像并命名为build-stage,在后续多阶段使用的时候有用

WORKDIR /app  # 设置当前目录

COPY package.json .   # 复制当前宿主机的package.json到 /app当前目录中

RUN npm config set registry https://registry.npmmirror.com/  # 设置npm 镜像源

RUN npm install # 安装依赖

COPY . . #  复制宿主机的也就是Dockfile 文件所在的目录到 /app 目录下

RUN npm run build  # 运行打包命令

FROM nginx:latest # 拉去nginx镜像

COPY --from=build-stage /app/dist /usr/share/nginx/html # 把上面build-stage阶段的内容也就是/app/dist目录下的内容复制到容器的/usr/share/nginx/html目录下

# 构建
docker build -t example .
# 启动
docker run -d -p 9999:80 example

# 在宿主机访问localhost:9999就可以成功访问
